# Icons

## Usage

```
<template>
  <Icon id="[iconId]" width="[width]" height="[height]">
</template>

<script>
import { FactorIcon } from '@mozilla/factor-ui';
import '@mozilla/factor-ui/dist/lib/factor.css';

export default {
  components: {
    FactorIcon
  },
}
</script>

```
